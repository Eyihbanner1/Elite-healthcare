'none' : 'block'; document.getElementById('nextBtn').textContent = step > totalSteps ? showStep(currentStep); } } function previousStep() { if (currentStep > 1) { currentStep--; showStep(currentStep); } } function validateStep(step) { const stepElement = document.getElementById('step' + step); } else if (careType === 'personal') { summary = 'Personal care services will help maintain dignity and comfort while ensuring daily needs are met safely.'; recommendations = [ { icon: 'fas fa-bath', text: 'Personal Hygiene Assistance - Bathing, grooming, and dressing support' }, { icon: 'fas fa-wheelchair', text: 'Mobility Support - Safe transfers and movement assistance' }, { icon: 'fas fa-pills', text: 'Medication Reminders - Ensuring medications are taken as prescribed' } ]; } else if (careType === 'medical') { summary = 'Professional medical care services are recommended to ensure health needs are properly managed.'; recommendations = [ { icon: 'fas fa-user-nurse', text: 'Skilled Nursing Care - Licensed nurses for medical monitoring' }, { icon: 'fas fa-heartbeat', text: 'Health Monitoring - Regular vital signs and health assessments' }, { icon: 'fas fa-syringe', text: 'Medical Procedures - Wound care, injections, and treatments' } ]; } else if (careType === 'specialized') { summary = 'Specialized care programs designed for complex needs with specially trained caregivers.'; recommendations = [ { icon: 'fas fa-brain', text: 'Memory Care Specialists - Trained in dementia and cognitive support' }, { icon: 'fas fa-hand-holding-heart', text: '24/7 Supervision - Round-the-clock care and monitoring' }, { icon: 'fas fa-puzzle-piece', text: 'Customized Care Plans - Tailored to specific medical conditions' } ]; } // Add location-specific recommendations if (location === 'home') { recommendations.push({ icon: 'fas fa-home', text: 'Home Safety Assessment - Ensuring environment is safe for care' }); } else if (location === 'facility') { recommendations.push({ icon: 'fas fa-building', text: 'Facility Placement Assistance - Help finding the right care community' });
